1. Lazy Mode Script (Battery)

#!/bin/bash
# ~/.config/eww/lazy-toggle.sh
SYSCTL_FILE="/etc/sysctl.conf"
TARGET_LINES=("vm.dirty_background_ratio" "vm.dirty_ratio")

# Determine current state
ENABLED=true
for LINE in "${TARGET_LINES[@]}"; do
  if ! grep -q "^\s*#\s*$LINE" "$SYSCTL_FILE"; then
    ENABLED=false
    break
  fi
done

if [ "$ENABLED" = true ]; then
  # Disable lazy mode = uncomment lines
  for LINE in "${TARGET_LINES[@]}"; do
    sudo sed -i "s|^\s*#\s*\($LINE.*\)|\1|" "$SYSCTL_FILE"
  done
  echo "Lazy Mode Disabled"
else
  # Enable lazy mode = comment lines
  for LINE in "${TARGET_LINES[@]}"; do
    sudo sed -i "s|^\($LINE.*\)|#\1|" "$SYSCTL_FILE"
  done
  echo "Lazy Mode Enabled"
fi

# Reload sysctl
sudo sysctl -p

2. Wi-Fi Toggle Script

#!/bin/bash
# ~/.config/eww/wifi-toggle.sh

if nmcli radio wifi | grep -q "enabled"; then
  nmcli radio wifi off
else
  nmcli radio wifi on
fi

4. Bluetooth Toggle Script

#!/bin/bash
# ~/.config/eww/bt-toggle.sh

if rfkill list bluetooth | grep -q "Soft blocked: no"; then
  rfkill block bluetooth
else
  rfkill unblock bluetooth
fi

5. Connect to Bluetooth Device Script

#!/bin/bash
# ~/.config/eww/bt-connect.sh
